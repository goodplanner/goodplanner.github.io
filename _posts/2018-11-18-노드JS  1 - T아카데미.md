## 비동기 함수모양

```

함수 구현
function add(i,j,callback) {  // 비동기함수는 값을 리턴을 하지 않고, 콜백함수를 호출해서 결과를 처리함.
  var result = i+j
  callback(result)
}

함수 사용

add(1,2,function(result) {
  console.log("Result:", result);
}




비동기 함수의 에러처리

callbackFunc(arg1, arc2, function(error, result) {   // 에러값, 결과값
  if(error) {
    // 에러처리
    return;
  }
  // 정상처리
}
```
